# 1. 启动 Ollama
pkill ollama 2>/dev/null; export OLLAMA_HOST=0.0.0.0:11434; export OLLAMA_CONTEXT_LENGTH=4096; export OLLAMA_NUM_THREADS=6; /Applications/Ollama.app/Contents/Resources/ollama serve &

# 2. 等待
sleep 5

# 3. 配置 Hermes
mkdir -p ~/.hermes && cat > ~/.hermes/config.yaml << 'EOF'
provider: custom
api_base: http://localhost:11434/v1
api_key: ollama
model: gemma4:e4b
max_context: 4096
max_tokens: 1024
EOF

# 4. 验证并启动
curl http://localhost:11434/api/tags && hermes
