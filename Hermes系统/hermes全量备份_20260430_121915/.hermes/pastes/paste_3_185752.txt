### 4.2 招投标公告自动检索（NAS定时任务）

```python
# bid_scraper.py - 运行在NAS上
import schedule
import time
from datetime import datetime
import requests
from bs4 import BeautifulSoup

class BidScraper:
    def __init__(self):
        self.sources = [
            "http://ggzy.zwfwb.tj.gov.cn",  # 天津公共资源
            "http://www.ccgp.gov.cn",        # 政府采购
            "https://www.chinabidding.com",  # 中国采购与招标
        ]
        self.ai_nas = OllamaClient("localhost:11434")
        
    def fetch_bids(self):
        """每日抓取招标公告"""
        bids = []
        for url in self.sources:
            try:
                resp = requests.get(url, timeout=30)
                soup = BeautifulSoup(resp.text, 'html.parser')
                items = soup.select('.bid-item')  # 根据实际结构调整
                bids.extend(items)
            except Exception as e:
                print(f"抓取失败 {url}: {e}")
        
        return bids
    
    def ai_filter(self, bids: list) -> list:
        """AI筛选相关公告"""
        keywords = ["建筑工程", "装饰装修", "设计施工一体化", "幕墙", "钢结构"]
        
        filtered = []
        for bid in bids:
            # 本地AI快速摘要
            summary = self.ai_nas.generate(
                model="qwen2.5:1.5b",
                prompt=f"摘要：{bid.text}\n\n判断是否相关：{keywords}",
                options={"num_predict": 100}
            )
            
            if "相关" in summary or "匹配" in summary:
                filtered.append({
                    "title": bid.title,
                    "date": bid.date,
                    "url": bid.link,
                    "summary": summary,
                    "match_score": self._score(summary)
                })
        
        return sorted(filtered, key=lambda x: x['match_score'], reverse=True)[:10]
    
    def notify(self, bids: list):
        """推送到微信/钉钉"""
        if not bids:
            return
        
        message = f"📋 今日招标 ({len(bids)}条)\n"
        for bid in bids[:5]:  # 前5条
            message += f"\n🔹 {bid['title'][:20]}...\n"
            message += f"   截止：{bid['date']} | 匹配度：{bid['match_score']}%\n"
            message += f"   {bid['url']}\n"
        
        # 发送到微信（使用Server酱等）
        requests.post(
            "https://sctapi.ftqq.com/{key}.send",
            data={"title": "招标提醒", "desp": message}
        )
    
    def run(self):
        bids = self.fetch_bids()
        filtered = self.ai_filter(bids)
        self.notify(filtered)

# 定时任务
schedule.every().day.at("09:00").do(BidScraper().run)
schedule.every().day.at("17:00").do(BidScraper().run)

while True:
    schedule.run_pending()
    time.sleep(60)
```

    def process_recording(self, audio_path: str, meeting_type: str) -> dict:
        """
        会议录音 -> 文字 -> 摘要 -> 待办
        """
        # 1. 语音转文字（本地，保护隐私）
        text = self.whisper.transcribe(audio_path)
        
        # 2. AI摘要（根据会议类型）
        if meeting_type == 'project':
            prompt = f"""工程项目会议纪要：
            {text}
            
            提取：
            1. 决议事项（带责任人）
            2. 进度节点（带日期）
            3. 风险预警
            4. 待办任务（自动创建日历）
            """
        elif meeting_type == 'design':
            prompt = f"""设计评审纪要：
            {text}
            
            提取：
            1. 修改意见（按专业分类）
            2. 规范依据
            3. 图纸变更清单
            """
        
        summary = self.ai.generate(model="qwen2.5:3b", prompt=prompt)
        
        # 3. 自动生成待办并入项目管理系统
        todos = self._extract_todos(summary)
        self._create_calendar_events(todos)
        
        return {
            "transcript": text,
            "summary": summary,
            "todos": todos,
            "storage": "公司NAS加密"
        }
